\textbf{Taylor series}
\fbox{\begin{minipage}{0.31\textwidth}
\begin{align*}
& f(x) = \sum_{k=0}^n \frac{f^{(k)}(x_0)}{k!} (x-x_0)^k + o((x-x_0)^n) \\
& \\
&	e^x = \sum_{n = 0}^\infty \frac{x^n}{n!} \hspace{22mm}
	\ln (1 + x) = \sum_{n = 1}^\infty (-1)^{n + 1} \frac{x^n}{n} \\
&	\cos x = \sum_{n=0}^\infty (-1)^n \frac{x^{2n}}{(2n)!} \hspace{5mm}
	\sin x = \sum_{n=0}^\infty (-1)^n \frac{x^{2n + 1}}{(2n + 1)!}
\end{align*}
\end{minipage}}

\textbf{Generating functions:}
\fbox{\begin{minipage}{0.31\textwidth}
\begin{align*}
& [x^i](1 + x)^n = C_{n}^{i} \hspace{18mm}
 [x^i](1 - x)^{-n} = C_{n + i - 1}^{i}\\
&C_{\alpha}^n = \frac{\alpha(\alpha - 1) \dots (\alpha - n + 1)}{n!}\\ 
& \prod_{n \ge 1} (1 - x^n) = \sum_{k = -\infty}^{\infty} (-1)^k x^{\frac{k(3k - 1)}{2}}
\end{align*}
\end{minipage}}

\textbf{Simpson's rule}\\
$n$ -- even number, $h = \frac{b - a}{n}$, $x_i = a + i h$
\scalebox{0.8}{
$\int_a^b f\Big(x\Big) \ud x \approx \frac{h}{3} \left[f\Big(x_0\Big) + 4 \sum_{i = 1}^{\frac{n}{2}} f\Big(x_{2 i - 1}\Big) + 
2 \sum_{i = 1}^{\frac{n}{2} - 1} f\Big(x_{2 i}\Big) + f\Big(x_n\Big) \right] $}

\textbf{Runge-Kutta 4th Order}
\fbox{\begin{minipage}{0.31\textwidth}
\begin{align*}
& \frac{\ud y}{\ud x} = f(x, y), \hspace{15mm} y(0) = y_0, \hspace{12mm} x_{i + 1} - x_i = h, \\
& y_{i + 1} = y_i + \frac{1}{6}(k_1 + 2 k_2 + 2 k_3 + k_4) h, \\
& k_1 = f(x_i, y_i), \hspace{21mm} k_2 = f(x_i + \frac{1}{2} h, y_i + \frac{1}{2} k_1 h), \\
& k_3 = f(x_i + \frac{1}{2} h, y_i + \frac{1}{2} k_2 h), \hspace{4mm} k_4 = f(x_i + h, y_i + k_3 h).
\end{align*}
\end{minipage}}

\textbf{List of integrals}
\fbox{\begin{minipage}{0.31\textwidth}
\begin{align*}
&    \int \frac{\ud x}{a^2 + x^2} = \frac{1}{a} \arctg \frac{x}{a} + C \\
&    \int \frac{\ud x}{a^2 - x^2} = \frac{1}{2a} \ln \left\lvert \frac{x + a}{x - a} \right\rvert + C \\
&    \int \frac{\ud x}{\sqrt{a^2 - x^2}} = \arcsin \frac{x}{a} + C \\
&    \int \frac{\ud x}{\sqrt{x^2 + a}} = \ln \left\lvert x + \sqrt{x^2 + a} \right\rvert + C \\
&    \int \frac{\ud x}{\cos^2 x} = \tg x + C \\
&    \int \frac{\ud x}{\sin^2 x} = -\ctg x + C
\end{align*}
\end{minipage}}

\textbf{Vandermonde matrix}\\
$V = V(x_0, x_1, \cdots, x_m) =
\begin{bmatrix}
1 & x_0 & x_0^2 & \dots & x_0^n\\
1 & x_1 & x_1^2 & \dots & x_1^n\\
1 & x_2 & x_2^2 & \dots & x_2^n\\
\vdots & \vdots & \vdots & \ddots &\vdots \\
1 & x_m & x_m^2 & \dots & x_m^n
\end{bmatrix}$

$$V_{i,j} = x_i^j, \hspace{5mm} \det(V)=\prod_{0\le i<j\le n} (x_j-x_i).$$ 

\textbf{Hadamard matrix}\\
$H_{1} = \begin{bmatrix}
     1
 \end{bmatrix}$, \hspace{10mm}
$H_{2^k} = \begin{bmatrix}
     H_{2^{k-1}} &  H_{2^{k-1}}\\
     H_{2^{k-1}} & -H_{2^{k-1}}
  \end{bmatrix}$\\
  
$det(H_n) = \pm n^{\frac{n}{2}}$\\

For matrix $M$ that $|M_{ij}| \le 1$ holds $|\operatorname{det}(M)| \leq n^{n/2}$.
