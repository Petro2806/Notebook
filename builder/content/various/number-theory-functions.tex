\subsection{Number theory functions}
\fbox{\begin{minipage}{0.31\textwidth}
\begin{align*}
& For \; n = p_1^{\alpha_1} \cdot \dots \cdot p_k^{\alpha_k}\\
& \phi(n) = \prod p_i^{\alpha_i - 1} (p_i - 1) - number \; of \; coprime \le n\\
& F(n) = \frac{n \cdot \phi(n)}{2} - sum \; of \; coprime \le n, \; for \; n > 1\\
& \mu(1) = 1\\
& \sigma_k(n) = \sum_{d|n} d^k\\
& \sigma_0(n) = \prod (\alpha_i + 1)\\
& \sigma_{k > 0}(n) = \prod \frac{p_i^{(\alpha_i + 1) \cdot k} - 1}{p_i^k - 1} 
\end{align*}
\end{minipage}}
